\documentclass{article}
\usepackage{algorithm2e}
\begin{document}

\begin{algorithm}
\KwData{unsorted \quad array \rightarrow arr, \quad $n \geq 0$}
\KwResult{sorted \quad array \rightarrow arr, \quad $n \geq 0$}
$j \gets 0$\;
$k \gets 1$\;
$q \gets n-1$\;
$sorted \gets true$\;
\While{$k < q$}
{
    \While{$j < q$}
    {
        \If{$arr[j] > arr[j+1]$}
        {
            swap $arr[j] \leftrightarrow arr[j+1]$
            
            $sorted \gets false$\;
        }

        $j \gets j + 1$\;
    }
    \If{$sorted == true$}
    {
        return $arr$;
    }
    $k \gets k + 1$\;
}
\end{algorithm}
